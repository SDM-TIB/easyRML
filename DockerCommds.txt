Build the image:
docker build -t sdmtib/easyrml:3.2 .

docker login -u="" -p=""

Push the image:
docker push sdmtib/easyrml:3.1

To remove previous image locally:
docker image rm sdmtib/easyrml:3.2

Pull and run the image:
docker run -d --name easyrml -p 5000:5000 sdmtib/easyrml:3.2

docker exec -it easyrml bash -c "..."
