Build the image:
docker build -t sdmtib/easyrml:3.1 .

docker login -u="" -p=""

Push the image:
docker push sdmtib/easyrml:3.1

To remove previous image locally:
docker image rm sdmtib/easyrml:3.0

Pull and run the image:
docker run -d --name easyrml -p 4200:4200 sdmtib/easyrml:3.1

docker exec -it easyrml bash -c "npm install @angular/cli --save-dev"
